```
function disemvowel(str) {
 let i = 0;
  let result = '';
  while (i < str.length) {

    if (str[i].toLowerCase() === 'a') {
      result = result + '';
    } else if (str[i].toLowerCase() === 'e') {
      result = result + '';
    } else if (str[i].toLowerCase() === 'i') {
      result = result + '';
    } else if (str[i].toLowerCase() === 'o') {
      result = result + '';
    } else if (str[i].toLowerCase() === 'u') {
      result = result + '';
    } else {
      result = result + str[i];
    }
    i = i + 1;
  }

  return result;
}
```
